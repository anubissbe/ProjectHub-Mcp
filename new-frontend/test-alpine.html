Test in ProjectHub

Open the browser console on http://localhost:8090 and run:

// Get the Alpine component
let app = document.querySelector('[x-data]').__x.$data;

// Check current view
console.log('Current view:', app.currentView);

// Switch to board
app.currentView = 'board';

// Switch to analytics  
app.currentView = 'analytics';

// Check if views exist
console.log('Board element:', document.querySelector('[x-show*="board"]'));
console.log('Analytics element:', document.querySelector('[x-show*="analytics"]'));
